<script>

function runSimulation() {

document.getElementById("formContainer").style.display = "none";

document.getElementById("loadingOverlay").style.display = "flex";

 

// Simulate calculation delay

setTimeout(() => {

calculateAndRender();

document.getElementById("loadingOverlay").style.display = "none";

document.getElementById("dashboardContainer").style.display = "block";

}, 1500);

}

 

function calculateAndRender() {

// Inputs

const type = document.getElementById("incidentType").value;

const industryMult = parseFloat(document.getElementById("industryType").value);

const size = document.getElementById("revenueBracket").value;

const records = document.getElementById("recordCount").value;

 

// 1. BASE LOGIC

let revenueMult = 1;

let baseOpsLoss = 0;

switch(size) {

case "Micro": revenueMult = 0.5; baseOpsLoss = 5000; break;

case "Small": revenueMult = 1.0; baseOpsLoss = 15000; break;

case "Mid": revenueMult = 2.5; baseOpsLoss = 45000; break;

case "Large": revenueMult = 6.0; baseOpsLoss = 150000; break;

}

 

let incidentLabel = "";

let duration = "";

let irHours = 0, legalHours = 0, ransom = 0, downtime = 0;

const irRate = 450;

const legalRate = 600;

 

switch(type) {

case "Ransomware":

incidentLabel = "Ransomware Attack";

downtime = 24; duration = "24 Days Downtime";

irHours = 180 * revenueMult; legalHours = 80 * revenueMult;

if(size === "Micro") ransom = 15000;

if(size === "Small") ransom = 125000;

if(size === "Mid") ransom = 550000;

if(size === "Large") ransom = 1800000;

break;

case "BEC":

incidentLabel = "Business Email Compromise";

downtime = 5; duration = "2-4 Weeks Analysis";

irHours = 60 * revenueMult; legalHours = 30 * revenueMult;

if(size === "Small") ransom = 75000; // Fraud loss

if(size === "Mid") ransom = 250000;

break;

case "DataBreach":

incidentLabel = "Data Exfiltration";

downtime = 2; duration = "3-6 Months Lifecycle";

irHours = 120 * revenueMult; legalHours = 150 * revenueMult;

break;

case "Insider":

incidentLabel = "Insider Threat";

downtime = 5; duration = "2 Months Investigation";

irHours = 100 * revenueMult; legalHours = 50 * revenueMult;

break;

case "DoS":

incidentLabel = "DDoS Attack";

downtime = 3; duration = "72 Hours";

irHours = 40 * revenueMult; legalHours = 10 * revenueMult;

break;

}

 

// 2. FINES & NOTIFICATION

let userCount = 0;

if(records === "Low") userCount = 500;

if(records === "Medium") userCount = 5000;

if(records === "High") userCount = 25000;

if(records === "Critical") userCount = 75000;

 

let fineRisk = 0;

const notifyCost = userCount * 25;

// Law 25 Risk Calc

if(type === "DataBreach" || type === "Ransomware") {

fineRisk = userCount * 15 * industryMult;

}

 

// 3. COSTS

const costIR = irHours * irRate;

const costForensics = (irHours * 0.5) * irRate;

const costRecovery = costIR * 0.4;

const totalTech = costIR + costForensics + costRecovery;

 

const costLegal = legalHours * legalRate;

const totalLegal = costLegal + fineRisk + notifyCost;

 

const costBizInt = downtime * baseOpsLoss * industryMult;

const costPR = (size === "Large" || size === "Mid") ? 50000 * industryMult : 5000;

const totalBiz = costBizInt + costPR + ransom;

 

const grandTotal = totalTech + totalLegal + totalBiz;

 

// 4. UPDATE DOM

document.getElementById("resIncidentType").innerText = incidentLabel.toUpperCase();

document.getElementById("resGrandTotal").innerText = formatCurrency(grandTotal);

document.getElementById("resDuration").innerText = `Est. Impact Duration: ${duration}`;

 

// 5. UPDATE BREAKDOWNS

// Tech

document.getElementById("techDetails").innerHTML = `

<div class="cost-row"><span>IR Services</span><span class="price">${formatCurrency(costIR)}</span></div>

<div class="cost-row"><span>Digital Forensics</span><span class="price">${formatCurrency(costForensics)}</span></div>

<div class="cost-row"><span>IT Recovery</span><span class="price">${formatCurrency(costRecovery)}</span></div>

<div class="subtotal" style="color:var(--accent-blue)"><span>TECHNICAL TOTAL</span><span>${formatCurrency(totalTech)}</span></div>

`;

 

// Legal

let fineLabel = (fineRisk > 0) ? formatCurrency(fineRisk) : "N/A";

document.getElementById("legalDetails").innerHTML = `

<div class="cost-row"><span>Counsel & Coach</span><span class="price">${formatCurrency(costLegal)}</span></div>

<div class="cost-row"><span>Notification</span><span class="price">${formatCurrency(notifyCost)}</span></div>

<div class="cost-row"><span>Reg. Fines (Risk)</span><span class="price">${fineLabel}</span></div>

<div class="subtotal" style="color:var(--accent-purple)"><span>LEGAL TOTAL</span><span>${formatCurrency(totalLegal)}</span></div>

`;

 

// Business

let ransomRow = (ransom > 0) ? `<div class="cost-row" style="color:#fca5a5"><span>Extortion/Fraud</span><span class="price">${formatCurrency(ransom)}</span></div>` : "";

document.getElementById("bizDetails").innerHTML = `

<div class="cost-row"><span>Operational Downtime</span><span class="price">${formatCurrency(costBizInt)}</span></div>

<div class="cost-row"><span>Crisis PR</span><span class="price">${formatCurrency(costPR)}</span></div>

${ransomRow}

<div class="subtotal" style="color:var(--accent-red)"><span>BUSINESS OPERATIONS IMPACT</span><span>${formatCurrency(totalBiz)}</span></div>

`;

 

// 6. UPDATE CHART

updateChart(totalTech, totalLegal, totalBiz, grandTotal);

}

 

function updateChart(tech, legal, biz, total) {

const pTech = (tech / total) * 100;

const pLegal = (legal / total) * 100;

const pBiz = (biz / total) * 100;

 

// Create Conic Gradient

// Blue (Tech) -> Purple (Legal) -> Red (Biz)

const chart = document.getElementById("donutChart");

chart.style.background = `conic-gradient(

var(--accent-blue) 0% ${pTech}%,

var(--accent-purple) ${pTech}% ${pTech + pLegal}%,

var(--accent-red) ${pTech + pLegal}% 100%

)`;

 

// Find largest factor

let maxVal = Math.max(pTech, pLegal, pBiz);

let label = "Technical";

if(maxVal === pLegal) label = "Legal";

if(maxVal === pBiz) label = "Business";

 

document.getElementById("topCostPercent").innerText = Math.round(maxVal) + "%";

document.querySelector(".donut-label").innerText = label + " Impact";

}

 

function formatCurrency(num) {

return "$" + Math.round(num).toLocaleString() + " CAD";

}

 

function resetSim() {

document.getElementById("dashboardContainer").style.display = "none";

document.getElementById("formContainer").style.display = "block";

}
